\noindent
\textbf{Шаг} $k$\textbf{:} сначала среда и нагрузка, затем расчёт CRAC по $T_{\mathrm{ret}}$ с \emph{предыдущего} шага; далее стойка и серверы; затем помещение и новый $T_{\mathrm{ret}}$ для шага $k{+}1$.

\noindent\textbf{Соответствие блокам (подробности в \texttt{mathematical\_model.tex}):}

\begin{align*}
  &\textbf{Стойка.}\quad
  T_{\mathrm{in},i} = T_{\mathrm{sup}} + \sum_{j=1}^{N} R_{ij}\,(T_{\mathrm{out},j} - T_{\mathrm{sup}}).
  \\[0.45em]
  &\textbf{Сервер.}\quad
  P_i = P_{\mathrm{idle}} + (P_{\mathrm{max}} - P_{\mathrm{idle}})\,u_i, \\[-0.1em]
  &\qquad T_{\mathrm{chip},i}^{+} = T_{\mathrm{chip},i} + \Delta t\,\frac{P_i - \dot Q_{\mathrm{air},i}}{C_{\mathrm{th}}}, \qquad
  T_{\mathrm{out},i} = T_{\mathrm{in},i} + \varepsilon\,(T_{\mathrm{chip},i} - T_{\mathrm{in},i}).
  \\[0.45em]
  &\textbf{CRAC.}\quad
  Q_{\mathrm{req}} = \dot m_{\mathrm{air}}\, c_p\, \max\bigl(0,\, T_{\mathrm{ret}} - T_{\mathrm{sup}}^{\mathrm{tgt}}\bigr), \quad
  Q_{\mathrm{avail}} = Q_{\mathrm{cap}}\, s_{\mathrm{fan}}, \\[-0.1em]
  &\qquad Q_{\mathrm{act}} = \min(Q_{\mathrm{req}},\, Q_{\mathrm{avail}})\ \text{(по режимам)}, \quad
  T_{\mathrm{sup}} = T_{\mathrm{ret}} - \dfrac{Q_{\mathrm{act}}}{\dot m_{\mathrm{air}}\, c_p}.
  \\[0.45em]
  &\textbf{Помещение.}\quad
  T_{\mathrm{room}}^{+} = T_{\mathrm{room}} + \frac{\Delta t}{C_{\mathrm{room}}}\bigl(Q_{\mathrm{racks}} + Q_{\mathrm{wall}} + Q_{\mathrm{mix}} - Q_{\mathrm{act}}\bigr), \\[-0.1em]
  &\qquad T_{\mathrm{ret}}^{\mathrm{tgt}} = 0{,}65\,\overline{T_{\mathrm{out}}} + 0{,}35\,T_{\mathrm{room}}
  \ \text{(далее сглаживание по $\alpha$).}
  \\[0.45em]
  &\textbf{Метрики.}\quad
  \mathrm{PUE} = \dfrac{P_{\mathrm{IT}} + P_{\mathrm{inf}}}{P_{\mathrm{IT}}}, \qquad
  r_{\mathrm{oh}} = \frac{1}{N}\sum_{i=1}^{N} \mathbf{1}\{T_{\mathrm{chip},i} > T_{\mathrm{thr}}\}.
\end{align*}

\vspace{0.3em}
\noindent\textit{Режимы охлаждения} \texttt{free} / \texttt{chiller} / \texttt{mixed}, COP$(T_{\mathrm{out}})$ и $P_{\mathrm{fan}}$ описаны в \texttt{mathematical\_model.tex}.
